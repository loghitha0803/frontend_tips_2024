 const selectedCitiesSunny = Object.keys(cityData).filter(cityKey => {
    const temperature = parseInt(cityData[cityKey].temperature)
    console.log(temperature)
    const humidity = parseInt(cityData[cityKey].humidity)
    console.log(humidity)
    const precipitation = parseInt(cityData[cityKey].precipitation)
    console.log(precipitation)

    return !isNaN(temperature) && !isNaN(humidity) && !isNaN(precipitation) &&
    temperature > 29 &&
    humidity < 50 &&
    precipitation >= 50
  })
  console.log(selectedCitiesSunny)
  const compareByTemperature = (a, b) => b.temperature - a.temperature
  const sortedCitiesSunny = selectedCitiesSunny.sort(compareByTemperature)
